document.addEventListener('DOMContentLoaded', function() {
    const checkboxes = document.querySelectorAll('.filter-checkbox');
    const paginationLinks = document.querySelectorAll('.page-link');
    const pages = document.querySelectorAll('.page-item');

    checkboxes.forEach(checkbox => {
      checkbox.addEventListener('change', filterProducts);
    });

    paginationLinks.forEach(link => {
      link.addEventListener('click', function(event) {
        event.preventDefault();
        const page = this.getAttribute('data-page');
        switchPage(page);
      });
    });

    function filterProducts() {
      const selectedFilters = Array.from(checkboxes)
        .filter(checkbox => checkbox.checked)
        .map(checkbox => checkbox.getAttribute('data-filter'));

      const products = document.querySelectorAll('.product-item');

      products.forEach(product => {
        const productClasses = Array.from(product.classList);
        const matches = selectedFilters.every(filter => productClasses.includes(filter));
        product.style.display = matches ? '' : 'none';
      });
    }

    function switchPage(page) {
        let currentPage = Array.from(pages).find(page => page.style.display !== 'none');
        if (!currentPage) {
          currentPage = pages[0];
        }
      
        let currentPageIndex = Array.from(pages).indexOf(currentPage);
        let newPageIndex;
      
        if (page === 'prev') {
          newPageIndex = currentPageIndex > 0 ? currentPageIndex - 1 : 0;
        } else if (page === 'next') {
          newPageIndex = currentPageIndex < pages.length - 1 ? currentPageIndex + 1 : pages.length - 1;
        } else {
          newPageIndex = parseInt(page) - 1;
        }
      
        currentPage.style.display = 'none';
      
        if (pages[newPageIndex]) {
          pages[newPageIndex].style.display = 'flex';
          updatePaginationLinks(newPageIndex);
        } else {
          console.error(`Page index ${newPageIndex} not found.`);
        }
      }
      

    function updatePaginationLinks(activePageIndex) {
        paginationLinks.forEach(link => link.parentElement.classList.remove('active'));
        paginationLinks[activePageIndex + 1].parentElement.classList.add('active');

        paginationLinks[0].parentElement.classList.toggle('disabled', activePageIndex === 0);
        paginationLinks[paginationLinks.length - 1].parentElement.classList.toggle('disabled', activePageIndex === pages.length - 1);
      }
  });
